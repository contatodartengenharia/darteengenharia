# SEO Component

## src/seo/SEO.tsx

```tsx id="0n3nh8"
import { Helmet } from 'react-helmet-async'

type Props = {
  title: string
  description: string
}

export function SEO({ title, description }: Props) {
  return (
    <Helmet>
      <title>{title}</title>

      <meta
        name=\"description\"
        content={description}
      />

      <meta property=\"og:title\" content={title} />
      <meta property=\"og:description\" content={description} />
    </Helmet>
  )
}
```
